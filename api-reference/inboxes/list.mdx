---
title: "List inboxes"
api: "GET /v1/inboxes"
description: "List all inboxes for the authenticated organization, across all domains."
---

<RequestExample>
```typescript TypeScript
import { CommuneClient } from 'commune-ai';

const commune = new CommuneClient({ apiKey: process.env.COMMUNE_API_KEY });

const result = await commune.inboxes.list();

for (const inbox of result.data) {
  console.log(inbox.address);  // support@mycompany.com
}
```
```python Python
from commune import CommuneClient

client = CommuneClient()

result = client.inboxes.list()

for inbox in result.data:
    print(inbox.address)
```

## Response

<ResponseField name="data" type="object[]">
  Array of inbox objects for your organization. Includes inboxes from all custom domains and the shared Commune domain.

  <Expandable title="Inbox object properties" defaultOpen>
    <ResponseField name="id" type="string">
      Unique inbox identifier (UUID).
    </ResponseField>
    <ResponseField name="localPart" type="string">
      The local part of the inbox address (before `@`).
    </ResponseField>
    <ResponseField name="address" type="string">
      Full email address of the inbox.
    </ResponseField>
    <ResponseField name="displayName" type="string">
      Display name shown in outbound `From` headers.
    </ResponseField>
    <ResponseField name="agent" type="object">
      <Expandable title="properties">
        <ResponseField name="name" type="string">Agent name.</ResponseField>
        <ResponseField name="id" type="string">Agent ID, if linked to an agent identity.</ResponseField>
        <ResponseField name="metadata" type="object">Arbitrary metadata attached to the agent.</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="webhook" type="object">
      <Expandable title="properties">
        <ResponseField name="endpoint" type="string">Webhook endpoint URL.</ResponseField>
        <ResponseField name="events" type="string[]">Subscribed event types.</ResponseField>
        <ResponseField name="secret" type="string">HMAC signing secret for webhook payload verification.</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="extractionSchema" type="object">
      Structured extraction configuration, if set. See [Set extraction schema](/api-reference/inboxes/update) for details.

      <Note>Structured extraction requires the **Business** plan or higher.</Note>
    </ResponseField>
    <ResponseField name="limits" type="object">
      <Expandable title="properties">
        <ResponseField name="emailsPerDay" type="number">Per-inbox daily outbound email limit override.</ResponseField>
        <ResponseField name="emailsPerHour" type="number">Per-inbox hourly outbound email limit override.</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="status" type="string">
      Inbox status. Typically `"active"`.
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      ISO 8601 creation timestamp.
    </ResponseField>
    <ResponseField name="domain_id" type="string">
      ID of the domain this inbox belongs to.
    </ResponseField>
    <ResponseField name="domain_name" type="string">
      Domain name (e.g. `mycompany.com`).
    </ResponseField>
  </Expandable>
</ResponseField>
